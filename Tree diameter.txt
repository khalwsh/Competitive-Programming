void dfs(int node,int parent,int &far_node,vector<int>adj[],int &max_dist,int counter){
    if(max_dist<counter){
        max_dist=counter;
        far_node=node;
    }
    for(auto &val:adj[node]){
        if(val==parent)continue;
        dfs(val,node,far_node,adj,max_dist,counter+1);
    }
}
